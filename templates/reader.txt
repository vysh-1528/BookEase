<script>
// READING PROGRESS
window.addEventListener('scroll', () => {
  const winScroll = document.body.scrollTop || document.documentElement.scrollTop;
  const height = document.documentElement.scrollHeight - document.documentElement.clientHeight;
  const scrolled = (winScroll / height) * 100;
  document.getElementById('progressBar').style.width = scrolled + '%';
});

// WORD COUNT & READ TIME
const content = document.getElementById('readerContent').textContent;
const words = content.trim().split(/\s+/).length;
const chars = content.length;
const readMins = Math.ceil(words / 200);
document.getElementById('wordCount').textContent = words.toLocaleString() + ' words';
document.getElementById('readTime').textContent = '~' + readMins + ' min read';
document.getElementById('charCount').textContent = chars.toLocaleString() + ' characters';

// FONT SIZE
function setFont(size) {
  const body = document.getElementById('readerBody');
  body.className = body.className.replace(/font-\w+/, '') + ' font-' + size;
  document.querySelectorAll('.ctrl-btn').forEach(b => {
    if (['sm','md','lg','xl'].some(s => b.textContent.includes('A') && b.onclick?.toString().includes(s))) {
      b.classList.remove('active');
    }
  });
}

// THEMES
function setTheme(theme) {
  const body = document.getElementById('readerBody');
  body.className = body.className.replace(/theme-\w+/, '') + ' theme-' + theme;
  ['light','dark','sepia'].forEach(t => {
    const btn = document.getElementById('theme' + t.charAt(0).toUpperCase() + t.slice(1));
    if (btn) btn.classList.toggle('active', t === theme);
  });
}

// TEXT TO SPEECH
let ttsActive = false;
let ttsUtterance = null;
function toggleTTS() {
  const btn = document.getElementById('ttsBtn');
  if (ttsActive) {
    window.speechSynthesis.cancel();
    ttsActive = false;
    btn.textContent = '🔊 Read';
    btn.classList.remove('active');
    return;
  }
  const text = document.getElementById('readerContent').textContent;
  ttsUtterance = new SpeechSynthesisUtterance(text.substring(0, 5000));
  ttsUtterance.rate = 0.9;
  ttsUtterance.onend = () => {
    ttsActive = false;
    btn.textContent = '🔊 Read';
    btn.classList.remove('active');
  };
  window.speechSynthesis.speak(ttsUtterance);
  ttsActive = true;
  btn.textContent = '⏹ Stop';
  btn.classList.add('active');
}
</script>
